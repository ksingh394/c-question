#include <stdio.h>
int main ()
{
int num,rem,rev=0,num_cpy;
printf("\nEnter the number :- ")
scanf("%d" , & num);
num_cpy=num;
for (num; num != 0 ; num/=10) //num = num/10
{
rem = num%10;
rev= rev*10+rem;
}
if ( num_cpy == rev )
printf("\n\t%d is the palindrome number ", num_cpy);
else
printf("\n\t%d is not the palindrom number ", num_cpy);
return 0 ;
}
