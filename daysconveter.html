#include <stdio.h>
 int main ()
{
int days,years,month,weeks,day;
printf ( "Enter days :- " ) ; // number of days given by user
scanf ( "%d" ,&days ) ;
// Converts days to years,months,weeks and days
years = days/365; // Calculate years
month = (days % 365)/30; // calculate months
weeks = ((days % 365)%30)/7; // Calculate weeks
day = days - ( ( years*365 ) + ( month*30 ) + ( weeks*7 ) ) ; // Calculate remaining day
// Print the results
printf ( "\tYear = %d\n\tMonth = %d\n\tWeek = %d\n\tDay = %d" ,years,month,weeks,day ) ;
 return 0 ;
 }
