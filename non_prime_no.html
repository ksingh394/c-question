#include <stdio.h>
int main ()
{
int num;
printf("\n\tEnter the number :- ");
scanf("%d " , &num);
printf("Non-prime numbers between 1 and %d are : " , num);

// Loop through each number from 2 to given number
for ( int i = 2; i <= num; i++)
{
int p = 0;
// Check if i is prime
for ( int j = 2; j < i; j++)
 {
if (i % j == 0 )
{
p++; // i is not prime
break ;
}
}

// If i is prime, print i
if ( p != 0 )
{
printf("%d ", i)
}

}

printf("\n\n")

return 0 ;
}
